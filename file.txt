    <script>
        lucide.createIcons();

        // --- AI Image Generation Logic ---
        
        const IMAGE_MODEL_NAME = 'imagen-3.0-generate-002';
        const apiKey = ""; // Leave as empty string for Canvas environment
        const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/${IMAGE_MODEL_NAME}:predict?key=${apiKey}`;

        const imagePromptEl = document.getElementById('imagePrompt');
        const generateBtn = document.getElementById('generateBtn');
        const imageLoader = document.getElementById('imageLoader');
        const generatedImage = document.getElementById('generatedImage');
        const imagePlaceholder = document.getElementById('imagePlaceholder');
        const imageError = document.getElementById('imageError');

        function showLoading(isLoading) {
            generateBtn.disabled = isLoading;
            imagePromptEl.disabled = isLoading;
            imageError.classList.add('hidden');
            // Hide/show the image and placeholder based on loading state
            if (isLoading) {
                generatedImage.classList.add('hidden');
                imagePlaceholder.classList.add('hidden');
            }
            
            if (isLoading) {
                imageLoader.classList.remove('hidden');
                generateBtn.innerHTML = '<div class="loader w-5 h-5 border-2 mr-2"></div> Generating...';
            } else {
                imageLoader.classList.add('hidden');
                generateBtn.innerHTML = '<i data-lucide="wand-2" class="w-5 h-5 mr-2"></i> Generate Image';
                // Re-render Lucide icons after changing button content
                lucide.createIcons();
            }
        }

        async function exponentialBackoffFetch(url, options, maxRetries = 5) {
            for (let attempt = 0; attempt < maxRetries; attempt++) {
                try {
                    const response = await fetch(url, options);
                    if (response.status === 429 && attempt < maxRetries - 1) {
                        // Exponential backoff with jitter: (2^attempt * 1000) + random_jitter
                        const delay = Math.pow(2, attempt) * 1000 + Math.random() * 1000;
                        await new Promise(resolve => setTimeout(resolve, delay));
                        continue; // Retry
                    }
                    if (!response.ok) {
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }
                    return response;
                } catch (error) {
                    if (attempt === maxRetries - 1) {
                        throw error;
                    }
                    // Exponential backoff with jitter
                    const delay = Math.pow(2, attempt) * 1000 + Math.random() * 1000;
                    await new Promise(resolve => setTimeout(resolve, delay));
                }
            }
        }

        async function generateImage() {
            const prompt = imagePromptEl.value.trim();
            if (!prompt) {
                imageError.textContent = "Please enter a descriptive prompt to generate an image.";
                imageError.classList.remove('hidden');
                imagePlaceholder.classList.remove('hidden');
                return;
            }

            showLoading(true);

            const payload = { 
                instances: { prompt: prompt }, 
                parameters: { "sampleCount": 1 } 
            };

            const options = {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            };

            try {
                const response = await exponentialBackoffFetch(apiUrl, options);
                const result = await response.json();

                if (result.predictions && result.predictions.length > 0 && result.predictions[0].bytesBase64Encoded) {
                    const base64Data = result.predictions[0].bytesBase64Encoded;
                    const imageUrl = `data:image/png;base64,${base64Data}`;
                    
                    generatedImage.src = imageUrl;
                    generatedImage.classList.remove('hidden');
                    imagePlaceholder.classList.add('hidden');

                } else {
                    const errorMessage = result.error?.message || "Failed to generate image. The model might have rejected the prompt, or an API error occurred.";
                    imageError.textContent = errorMessage;
                    imageError.classList.remove('hidden');
                    imagePlaceholder.classList.add('hidden');
                    console.error("API response error:", result);
                }

            } catch (error) {
                imageError.textContent = "A connection error occurred. Please check the console for details.";
                imageError.classList.remove('hidden');
                imagePlaceholder.classList.add('hidden');
                console.error("Fetch error:", error);
            } finally {
                showLoading(false);
            }
        }

            if (footerText) {
                footerText.innerHTML = footerText.innerHTML.replace('[Your Name]', 'Aqsa Mahak');
            }
        });
